function solution(n) {
  var answer = 0;
  let x = n + 1;
  const cntN = countOne(n);

  while (x) {
    const cntX = countOne(x);
    if (cntN === cntX){return x;}
    x++;
  }
  return answer;
}


function countOne(value){
    return value.toString(2).split("").filter((v) => v === "1").length;
}
